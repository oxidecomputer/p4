32/// An object for keeping track of state as we generate code.
33#[derive(Default)]
34struct Context {
35    /// Rust structs we've generated.
36    structs: HashMap<String, TokenStream>,
37
38    /// Rust functions we've generated.
39    functions: HashMap<String, TokenStream>,
40
41    /// Pipeline structures we've generated.
42    pipelines: HashMap<String, TokenStream>,
43}

50pub struct Sanitizer {}

52impl Sanitizer {
53    pub fn sanitize_string(s: &mut String) {
54        //TODO sanitize other problematic rust tokens
55        if s == "type" {
56            "typ".clone_into(s)
57        }
58        if s == "match" {
59            "match_".clone_into(s)
60        }
61    }
62}
63
64impl MutVisitor for Sanitizer {
65    fn struct_member(&self, m: &mut StructMember) {
66        Self::sanitize_string(&mut m.name);
67    }
68    fn header_member(&self, m: &mut HeaderMember) {
69        Self::sanitize_string(&mut m.name);
70    }
71    fn control_parameter(&self, p: &mut ControlParameter) {
72        Self::sanitize_string(&mut p.name);
73    }
74    fn action_parameter(&self, p: &mut ActionParameter) {
75        Self::sanitize_string(&mut p.name);
76    }
77    fn lvalue(&self, lv: &mut Lvalue) {
78        Self::sanitize_string(&mut lv.name);
79    }
80}

82pub fn sanitize(ast: &mut AST) {
83    let s = Sanitizer {};
84    ast.mut_accept(&s);
85}

87pub fn emit(
88    ast: &AST,
89    hlir: &Hlir,
90    filename: &str,
91    settings: Settings,
92) -> io::Result<()> {
93    let tokens = emit_tokens(ast, hlir, settings);
94
95    //
96    // format the code and write it out to a Rust source file
97    //
98    let f: syn::File = match syn::parse2(tokens.clone()) {
99        Ok(f) => f,
100        Err(e) => {
101            // On failure write generated code to a tempfile
102            println!("Code generation produced unparsable code");
103            write_to_tempfile(&tokens)?;
104            return Err(io::Error::other(format!(
105                "Failed to parse generated code: {:?}",
106                e
107            )));
108        }
109    };
110    fs::write(filename, prettyplease::unparse(&f))?;
111
112    Ok(())
113}
114
115fn write_to_tempfile(tokens: &TokenStream) -> io::Result<()> {
116    let mut out = tempfile::Builder::new().suffix(".rs").tempfile()?;
117    out.write_all(tokens.to_string().as_bytes())?;
118    println!("Wrote generated code to {}", out.path().display());
119    out.keep()?;
120    Ok(())
121}

123pub fn emit_tokens(ast: &AST, hlir: &Hlir, settings: Settings) -> TokenStream {
124    //
125    // initialize a context to track state while we generate code
126    //
127
128    let mut ctx = Context::default();
129
130    //
131    // genearate rust code for the P4 AST
132    //
133
134    let mut hg = HeaderGenerator::new(ast, &mut ctx);
135    hg.generate();
136
137    let mut sg = StructGenerator::new(ast, &mut ctx);
138    sg.generate();
139
140    let mut pg = ParserGenerator::new(ast, hlir, &mut ctx);
141    pg.generate();
142
143    let mut cg = ControlGenerator::new(ast, hlir, &mut ctx);
144    cg.generate();
145
146    let mut pg = PipelineGenerator::new(ast, hlir, &mut ctx, &settings);
147    pg.generate();
148
149    //
150    // collect all the tokens we generated into one stream
151    //
152
153    // start with use statements
154    let mut tokens = quote! {
155        use p4rs::{checksum::Checksum, *};
156        use colored::*;
157        use bitvec::prelude::*;
158    };
159
160    //to lib dtrace probes
161    tokens.extend(dtrace_probes());
162
163    // structs
164    for s in ctx.structs.values() {
165        tokens.extend(s.clone());
166    }
167
168    // functions
169    for s in ctx.functions.values() {
170        tokens.extend(s.clone());
171    }
172
173    // pipelines
174    for p in ctx.pipelines.values() {
175        tokens.extend(p.clone());
176    }
177
178    tokens
179}

183        #[usdt::provider]
184        mod softnpu_provider {
185            fn parser_accepted(_: &str) {}
186            fn parser_transition(_: &str) {}
187            fn parser_dropped() {}
188            fn control_apply(_: &str) {}
189            fn control_table_hit(_: &str) {}
190            fn control_table_miss(_: &str) {}
191            fn ingress_dropped(_: &str) {}
192            fn ingress_accepted(_: &str) {}
193            fn egress_dropped(_: &str) {}
194            fn egress_accepted(_: &str) {}
195            fn egress_table_hit(_: &str) {}
196            fn egress_table_miss(_: &str) {}
197            fn action(_: &str) {}
198        }
199    }

210/// Return the rust type for a given P4 type.
211fn rust_type(ty: &Type) -> TokenStream {
212    match ty {
213        Type::Bool => quote! { bool },
214        Type::Error => todo!("generate error type"),
215        Type::Bit(_size) => {
216            quote! { BitVec::<u8, Msb0> }
217        }
218        Type::Int(_size) => todo!("generate int type"),
219        Type::Varbit(_size) => todo!("generate varbit type"),
220        Type::String => quote! { String },
221        Type::UserDefined(name) => {
222            let typename = format_ident!("{}", name);
223            quote! { #typename }
224        }
225        Type::ExternFunction => {
226            todo!("rust type for extern function");
227        }
228        Type::HeaderMethod => {
229            todo!("rust type for header method");
230        }
231        Type::Table => {
232            todo!("rust type for table");
233        }
234        Type::Void => {
235            quote! { () }
236        }
237        Type::List(_) => todo!("rust type for list"),
238        Type::State => {
239            todo!("rust type for state");
240        }
241        Type::Action => {
242            todo!("rust type for action");
243        }
244    }

247fn type_size(ty: &Type, ast: &AST) -> usize {
248    match ty {
249        Type::Bool => 1,
250        Type::Error => todo!("generate error size"),
251        Type::Bit(size) => *size,
252        Type::Int(size) => *size,
253        Type::Varbit(size) => *size,
254        Type::String => todo!("generate string size"),
255        Type::UserDefined(name) => {
256            let mut sz: usize = 0;
257            let udt = ast.get_user_defined_type(name).unwrap_or_else(|| {
258                panic!("expect user defined type: {}", name)
259            });
260
261            match udt {
262                UserDefinedType::Struct(s) => {
263                    for m in &s.members {
264                        sz += type_size(&m.ty, ast);
265                    }
266                    sz
267                }
268                UserDefinedType::Header(h) => {
269                    for m in &h.members {
270                        sz += type_size(&m.ty, ast);
271                    }
272                    sz
273                }
274                UserDefinedType::Extern(_) => {
275                    todo!("size for extern?");
276                }
277            }
278        }
279        Type::ExternFunction => {
280            todo!("type size for extern function");
281        }
282        Type::HeaderMethod => {
283            todo!("type size for header method");
284        }
285        Type::Table => {
286            todo!("type size for table");
287        }
288        Type::Void => 0,
289        Type::List(_) => todo!("type size for list"),
290        Type::State => {
291            todo!("type size for state");
292        }
293        Type::Action => {
294            todo!("type size for action");
295        }
296    }
297}

299fn type_size_bytes(ty: &Type, ast: &AST) -> usize {
300    let s = type_size(ty, ast);
301    let mut b = s >> 3;
302    if !s.is_multiple_of(8) {
303        b += 1
304    }
305    b
306}

308// in the case of an expression
309//
310//   a &&& b
311//
312// where b is an integer literal interpret b as a prefix mask based on the
313// number of leading ones
314fn try_extract_prefix_len(expr: &Expression) -> Option<u8> {
315    match &expr.kind {
316        ExpressionKind::Binary(_lhs, _op, rhs) => match &rhs.kind {
317            ExpressionKind::IntegerLit(v) => Some(v.leading_ones() as u8),
318            ExpressionKind::BitLit(_width, v) => Some(v.leading_ones() as u8),
319            ExpressionKind::SignedLit(_width, v) => {
320                Some(v.trailing_ones() as u8)
321            }
322            _ => None,
323        },
324        _ => None,
325    }
326}

341fn is_header_member(lval: &Lvalue, hlir: &Hlir) -> bool {
342    if lval.degree() >= 1 {
343        let name_info = hlir
344            .lvalue_decls
345            .get(lval)
346            .unwrap_or_else(|| panic!("name for lval {:#?}", lval));
347
348        matches!(name_info.decl, DeclarationInfo::HeaderMember)
349    } else {
350        false
351    }
352}

355fn is_rust_reference(lval: &Lvalue, names: &HashMap<String, NameInfo>) -> bool {
356    if lval.degree() == 1 {
357        let name_info = names
358            .get(lval.root())
359            .unwrap_or_else(|| panic!("name for lval {:#?}", lval));
360
361        match name_info.decl {
362            DeclarationInfo::Parameter(Direction::Unspecified) => false,
363            DeclarationInfo::Parameter(Direction::In) => false,
364            DeclarationInfo::Parameter(Direction::Out) => true,
365            DeclarationInfo::Parameter(Direction::InOut) => true,
366            DeclarationInfo::Local => false,
367            DeclarationInfo::Method => false,
368            DeclarationInfo::StructMember => false,
369            DeclarationInfo::HeaderMember => false,
370            DeclarationInfo::ControlTable => false,
371            DeclarationInfo::ControlMember => false,
372            DeclarationInfo::State => false,
373            DeclarationInfo::Action => false,
374            DeclarationInfo::ActionParameter(_) => false,
375        }
376    } else {
377        false
378    }
379}

407fn table_qname(
408    chain: &Vec<(String, &Control)>,
409    table: &Table,
410    sep: char,
411) -> String {
412    let mut qname = String::new();
413    for c in chain {
414        if c.0.is_empty() {
415            continue;
416        }
417        qname += &format!("{}{}", c.0, sep);
418    }
419    qname += &table.name;
420    qname
