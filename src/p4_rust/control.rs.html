18pub(crate) struct ControlGenerator<'a> {
19    ast: &'a AST,
20    ctx: &'a mut Context,

24impl<'a> ControlGenerator<'a> {
25    pub(crate) fn new(
26        ast: &'a AST,
27        hlir: &'a Hlir,
28        ctx: &'a mut Context,
29    ) -> Self {
30        Self { ast, hlir, ctx }
31    }

33    pub(crate) fn generate(&mut self) {
34        for control in &self.ast.controls {
35            self.generate_control(control);
36        }
37
38        if let Some(ingress) = self.ast.get_control("ingress") {
39            self.generate_top_level_control(ingress);
40        };
41
42        if let Some(egress) = self.ast.get_control("egress") {
43            self.generate_top_level_control(egress);
44        };
45    }
46
47    fn generate_top_level_control(&mut self, control: &Control) {
48        let tables = control.tables(self.ast);
49        for (cs, t) in tables {
50            let qtn = qualified_table_function_name(Some(control), &cs, t);
51            let qtfn = qualified_table_function_name(Some(control), &cs, t);
52            let control = cs.last().unwrap().1;
53            let (_, mut param_types) = self.control_parameters(control);
54            for var in &control.variables {
55                if let Type::UserDefined(typename) = &var.ty {
56                    if self.ast.get_extern(typename).is_some() {
57                        let extern_type = format_ident!("{}", typename);
58                        param_types.push(quote! {
59                            &p4rs::externs::#extern_type
60                        })
61                    }
62                }
63            }
64            let (type_tokens, table_tokens) =
65                self.generate_control_table(control, t, &param_types);
66            let qtn = format_ident!("{}", qtn);
67            let qtfn = format_ident!("{}", qtfn);
68            self.ctx.functions.insert(
69                qtn.to_string(),
70                quote! {
71                    pub fn #qtfn() -> #type_tokens {
72                        #table_tokens
73                    }
74                },
75            );
76        }
77    }

79    pub(crate) fn generate_control(
80        &mut self,
81        control: &Control,
82    ) -> (TokenStream, TokenStream) {
83        let (mut params, _param_types) = self.control_parameters(control);
84
85        for action in &control.actions {
86            if action.name == "NoAction" {
87                continue;
88            }
89            self.generate_control_action(control, action);
90        }
91
92        let tables = control.tables(self.ast);
93        for (cs, table) in tables {
94            let c = cs.last().unwrap().1;
95            let qtn = qualified_table_function_name(None, &cs, table);
96            let (_, mut param_types) = self.control_parameters(c);
97            for var in &c.variables {
98                if let Type::UserDefined(typename) = &var.ty {
99                    if self.ast.get_extern(typename).is_some() {
100                        let extern_type = format_ident!("{}", typename);
101                        param_types.push(quote! {
102                            &p4rs::externs::#extern_type
103                        })
104                    }
105                }
106            }
107            let n = table.key.len();
108            let table_type = quote! {
109                p4rs::table::Table::<
110                    #n,
111                    std::sync::Arc<dyn Fn(#(#param_types),*)>
112                    >
113            };
114            let qtn = format_ident!("{}", qtn);
115            params.push(quote! {
116                #qtn: &#table_type
117            });
118        }
119
120        let name = format_ident!("{}_apply", control.name);
121        let apply_body = self.generate_control_apply_body(control);
122        let sig = quote! {
123            (#(#params),*)
124        };
125        self.ctx.functions.insert(
126            name.to_string(),
127            quote! {
128                pub fn #name #sig {
129                    #apply_body
130                }
131            },
132        );
133
134        (sig, apply_body)
135    }

137    pub(crate) fn control_parameters(
138        &mut self,
139        control: &Control,
140    ) -> (Vec<TokenStream>, Vec<TokenStream>) {
141        let mut params = Vec::new();
142        let mut types = Vec::new();
143
144        for arg in &control.parameters {
145            match arg.ty {
146                Type::UserDefined(ref typename) => {
147                    match self.ast.get_user_defined_type(typename) {
148                        Some(_udt) => {
149                            let name = format_ident!("{}", arg.name);
150                            let ty = rust_type(&arg.ty);
151                            match &arg.direction {
152                                Direction::Out | Direction::InOut => {
153                                    params.push(quote! {
154                                        #name: &mut #ty
155                                    });
156                                    types.push(quote! { &mut #ty });
157                                }
158                                _ => {
159                                    params.push(quote! {
160                                        #name: &#ty
161                                    });
162                                    types.push(quote! { &#ty });
163                                }
164                            }
165                        }
166                        None => {
167                            // if this is a generic type, skip for now
168                            if control.is_type_parameter(typename) {
169                                continue;
170                            }
171                            panic!("Undefined type {}", typename);
172                        }
173                    }
174                }
175                _ => {
176                    let name = format_ident!("{}", arg.name);
177                    let ty = rust_type(&arg.ty);
178                    match &arg.direction {
179                        Direction::Out | Direction::InOut => {
180                            params.push(quote! { #name: &mut #ty });
181                            types.push(quote! { &mut #ty });
182                        }
183                        _ => {
184                            if arg.ty == Type::Bool {
185                                params.push(quote! { #name: #ty });
186                                types.push(quote! { #ty });
187                            } else {
188                                params.push(quote! { #name: &#ty });
189                                types.push(quote! { &#ty });
190                            }
191                        }
192                    }
193                }
194            }
195        }
196
197        (params, types)
198    }

200    fn generate_control_action(&mut self, control: &Control, action: &Action) {
201        let name = format_ident!("{}_action_{}", control.name, action.name);
202        let (mut params, _) = self.control_parameters(control);
203
204        for var in &control.variables {
205            if let Type::UserDefined(typename) = &var.ty {
206                if self.ast.get_extern(typename).is_some() {
207                    let name = format_ident!("{}", var.name);
208                    let extern_type = format_ident!("{}", typename);
209                    params.push(quote! {
210                        #name: &p4rs::externs::#extern_type
211                    })
212                }
213            }
214        }
215
216        let mut dump_fmt = Vec::new();
217        for p in &action.parameters {
218            dump_fmt.push(p.name.clone() + "={}");
219        }
220        //let dump_fmt = vec!["{}"; action.parameters.len()];
221        let dump_fmt = dump_fmt.join(", ");
222        let dump_args: Vec<TokenStream> = action
223            .parameters
224            .iter()
225            .map(|x| format_ident!("{}", x.name.clone()))
226            .map(|x| quote! { #x })
227            .collect();
228
229        let dump = quote! {
230            // TODO find a way to only allocate the string when probe is active.
231            // Cannot simply return reference to string created within probe.
232            let dump = format!(#dump_fmt, #(#dump_args,)*);
233            softnpu_provider::action!(|| (&dump));
234        };
235
236        for arg in &action.parameters {
237            // if the type is user defined, check to ensure it's defined
238            if let Type::UserDefined(ref typename) = arg.ty {
239                match self.ast.get_user_defined_type(typename) {
240                    Some(_) => {
241                        let name = format_ident!("{}", arg.name);
242                        let ty = rust_type(&arg.ty);
243                        params.push(quote! { #name: #ty });
244                    }
245                    None => {
246                        panic!(
247                            "codegen: undefined type {} for arg {:#?}",
248                            typename, arg,
249                        );
250                    }
251                }
252            } else {
253                let name = format_ident!("{}", arg.name);
254                let ty = rust_type(&arg.ty);
255                params.push(quote! { #name: #ty });
256            }
257        }
258
259        let mut names = control.names();
260        let sg = StatementGenerator::new(
261            self.ast,
262            self.hlir,
263            StatementContext::Control(control),
264        );
265        let body = sg.generate_block(&action.statement_block, &mut names);
266
267        let __name = name.to_string();
268
269        self.ctx.functions.insert(
270            name.to_string(),
271            quote! {
272                pub fn #name(#(#params),*) {
273
274                    //TODO <<<< DTRACE <<<<<<
275                    //Generate dtrace prbes that allow us to trace control
276                    //action flows.
277                    //println!("####{}####", #__name);
278
279                    #dump
280
281                    #body
282                }
283            },
284        );
285    }

287    pub(crate) fn generate_control_table(
288        &mut self,
289        control: &Control,
290        table: &Table,
291        control_param_types: &Vec<TokenStream>,
292    ) -> (TokenStream, TokenStream) {
293        let mut key_type_tokens: Vec<TokenStream> = Vec::new();
294        let mut key_types: Vec<Type> = Vec::new();
295
296        for (k, _) in &table.key {
297            let parts: Vec<&str> = k.name.split('.').collect();
298            let root = parts[0];
299
300            // try to find the root of the key as an argument to the control block.
301            // TODO: are there other places to look for this?
302            match Self::get_control_arg(control, root) {
303                Some(_param) => {
304                    if parts.len() > 1 {
305                        let tm = control.names();
306                        //TODO: use hlir?
307                        let ty = resolve_lvalue(k, self.ast, &tm).unwrap().ty;
308                        key_types.push(ty.clone());
309                        key_type_tokens.push(rust_type(&ty));
310                    }
311                }
312                None => {
313                    panic!("bug: control arg undefined {:#?}", root)
314                }
315            }
316        }
317
318        let table_name = format_ident!("{}_table", table.name);
319        let n = table.key.len();
320        let table_type = quote! {
321            p4rs::table::Table::<
322                #n,
323                std::sync::Arc<dyn Fn(#(#control_param_types),*)>
324            >
325        };
326
327        let mut tokens = quote! {
328            let mut #table_name: #table_type = #table_type::new();
329        };
330
331        if table.const_entries.is_empty() {
332            tokens.extend(quote! { #table_name });
333            return (table_type, tokens);
334        }
335
336        for entry in &table.const_entries {
337            let mut keyset = Vec::new();
338            for (i, k) in entry.keyset.iter().enumerate() {
339                match &k.value {
340                    KeySetElementValue::Expression(e) => {
341                        let eg = ExpressionGenerator::new(self.hlir);
342                        let xpr = eg.generate_expression(e.as_ref());
343                        let ks = match table.key[i].1 {
344                            MatchKind::Exact => {
345                                let k = format_ident!("{}", "Exact");
346                                quote! {
347                                    p4rs::table::Key::#k(
348                                        p4rs::bitvec_to_biguint(&#xpr))
349                                }
350                            }
351                            MatchKind::Ternary => {
352                                let k = format_ident!("{}", "Ternary");
353                                quote! {
354                                    p4rs::table::Key::#k(
355                                        p4rs::bitvec_to_biguint(&#xpr))
356                                }
357                            }
358                            MatchKind::LongestPrefixMatch => {
359                                let len = match try_extract_prefix_len(e) {
360                                    Some(len) => len,
361                                    None => {
362                                        panic!(
363                                            "codegen: coult not determine prefix 
364                                        len for key {:#?}",
365                                        table.key[i].1,
366                                        );
367                                    }
368                                };
369                                let k = format_ident!("{}", "Lpm");
370                                quote! {
371                                    p4rs::table::Key::#k(p4rs::table::Prefix{
372                                        addr: bitvec_to_ip6addr(&(#xpr)),
373                                        len: #len,
374                                    })
375                                }
376                            }
377                            MatchKind::Range => {
378                                let k = format_ident!("Range");
379                                quote! {
380                                    p4rs::table::Key::#k(#xpr)
381                                }
382                            }
383                        };
384                        keyset.push(ks);
385                    }
386                    x => todo!("key set element {:?}", x),
387                }
388            }
389
390            let action = match control.get_action(&entry.action.name) {
391                Some(action) => action,
392                None => {
393                    panic!("codegen: action {} not found", entry.action.name);
394                }
395            };
396
397            let mut action_fn_args = Vec::new();
398            for arg in &control.parameters {
399                let a = format_ident!("{}", arg.name);
400                action_fn_args.push(quote! { #a });
401            }
402
403            let action_fn_name =
404                format_ident!("{}_action_{}", control.name, entry.action.name);
405            for (i, expr) in entry.action.parameters.iter().enumerate() {
406                match &expr.kind {
407                    ExpressionKind::IntegerLit(v) => {
408                        match &action.parameters[i].ty {
409                            Type::Bit(n) => {
410                                if *n <= 8 {
411                                    let v = *v as u8;
412                                    action_fn_args.push(quote! {
413                                        #v.view_bits::<Msb0>().to_bitvec()
414                                    });
415                                }
416                            }
417                            x => {
418                                todo!("action int lit expression type {:?}", x)
419                            }
420                        }
421                    }
422                    ExpressionKind::BitLit(width, v) => {
423                        match &action.parameters[i].ty {
424                            Type::Bit(n) => {
425                                let n = *n;
426                                if n != *width as usize {
427                                    panic!(
428                                        "{:?} not compatible with {:?}",
429                                        expr.kind, action.parameters[i],
430                                    );
431                                }
432                                let size = n;
433                                action_fn_args.push(quote! {{
434                                    let mut x = bitvec![mut u8, Msb0; 0; #size];
435                                    x.store_le(#v);
436                                    x
437                                }});
438                            }
439                            x => {
440                                todo!("action bit lit expression type {:?}", x)
441                            }
442                        }
443                    }
444                    x => todo!("action parameter type {:?}", x),
445                }
446            }
447
448            let mut closure_params = Vec::new();
449            for x in &control.parameters {
450                let name = format_ident!("{}", x.name);
451                closure_params.push(quote! { #name });
452            }
453
454            tokens.extend(quote! {
455
456                let action: std::sync::Arc<dyn Fn(#(#control_param_types),*)> =
457                    std::sync::Arc::new(|#(#closure_params),*| {
458                        #action_fn_name(#(#action_fn_args),*);
459                    });
460
461                #table_name.entries.insert(
462                    p4rs::table::TableEntry::<
463                        #n,
464                        std::sync::Arc<dyn Fn(#(#control_param_types),*)>,
465                    >{
466                        key: [#(#keyset),*],
467                        priority: 0,
468                        name: "your name here".into(),
469                        action,
470
471                        //TODO actual data, does this actually matter for
472                        //constant entries?
473                        action_id: String::new(),
474                        parameter_data: Vec::new(),
475                    });
476            })
477        }
478
479        tokens.extend(quote! { #table_name });
480
481        (table_type, tokens)
482    }

484    fn generate_control_apply_body(
485        &mut self,
486        control: &Control,
487    ) -> TokenStream {
488        let mut tokens = TokenStream::new();
489
490        for var in &control.variables {
491            //TODO check in checker that externs are actually defined by
492            //SoftNPU.
493            if let Type::UserDefined(typename) = &var.ty {
494                if self.ast.get_extern(typename).is_some() {
495                    let name = format_ident!("{}", var.name);
496                    let extern_type = format_ident!("{}", typename);
497                    tokens.extend(quote! {
498                        let #name = p4rs::externs::#extern_type::new();
499                    })
500                }
501            }
502        }
503
504        let mut names = control.names();
505        let sg = StatementGenerator::new(
506            self.ast,
507            self.hlir,
508            StatementContext::Control(control),
509        );
510        tokens.extend(sg.generate_block(&control.apply, &mut names));
511
512        tokens
513    }

515    fn get_control_arg<'b>(
516        control: &'b Control,
517        arg_name: &str,
518    ) -> Option<&'b ControlParameter> {
519        control.parameters.iter().find(|&arg| arg.name == arg_name)
520    }
