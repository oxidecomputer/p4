17// TODO transition from BigUint to BitVec<u8, Msb0>, this requires being able to
18// do a number of mathematical operations on BitVec<u8, Msb0>.
19#[derive(Debug, Clone, PartialEq, Hash, Eq, Serialize, Deserialize)]
20pub enum Key {
21    Exact(BigUintKey),
22    Range(BigUintKey, BigUintKey),
23    Ternary(Ternary),
24    Lpm(Prefix),
25}
26
27impl Default for Key {
28    fn default() -> Self {
29        Self::Ternary(Ternary::default())
30    }
31}

33impl Key {
34    pub fn to_bytes(&self) -> Vec<u8> {
35        match self {
36            Key::Exact(x) => {
37                let mut buf = x.value.to_bytes_le();
38
39                // A value serialized from a BigUint may be less than the width of a
40                // field. For example a 16-bit field with with a value of 47 will come
41                // back in 8 bits from BigUint serialization.
42                buf.resize(x.width, 0);
43                buf
44            }
45            Key::Range(a, z) => {
46                let mut buf_a = a.value.to_bytes_le();
47                let mut buf_z = z.value.to_bytes_le();
48
49                buf_a.resize(a.width, 0);
50                buf_z.resize(z.width, 0);
51                buf_a.extend_from_slice(&buf_z);
52                buf_a
53            }
54            Key::Ternary(t) => match t {
55                Ternary::DontCare => {
56                    let mut buf = Vec::new();
57                    buf.clear();
58                    buf
59                }
60                Ternary::Value(v) => {
61                    let mut buf = v.value.to_bytes_le();
62                    buf.resize(v.width, 0);
63                    buf
64                }
65                Ternary::Masked(v, m, w) => {
66                    let mut buf_a = v.to_bytes_le();
67                    let mut buf_b = m.to_bytes_le();
68                    buf_a.resize(*w, 0);
69                    buf_b.resize(*w, 0);
70                    buf_a.extend_from_slice(&buf_b);
71                    buf_a
72                }
73            },
74            Key::Lpm(p) => {
75                let mut v: Vec<u8> = match p.addr {
76                    IpAddr::V4(a) => a.octets().into(),
77                    IpAddr::V6(a) => a.octets().into(),
78                };
79                v.push(p.len);
80                v
81            }
82        }
83    }
84}

112impl<const D: usize, A: Clone> Table<D, A> {
113    pub fn new() -> Self {
114        Self {
115            entries: HashSet::new(),
116        }
117    }
118
119    pub fn match_selector(
120        &self,
121        keyset: &[BigUint; D],
122    ) -> Vec<TableEntry<D, A>> {
123        let mut result = Vec::new();
124        for entry in &self.entries {
125            if keyset_matches(keyset, &entry.key) {
126                result.push(entry.clone());
127            }
128        }
129        sort_entries(result)
130    }
131
132    pub fn dump(&self) -> String {
133        let mut s = String::new();
134        for e in &self.entries {
135            writeln!(s, "{:?}", e.key).unwrap();
136        }
137        s
138    }
139}

141// Sort a vector of match result entries. Do not apply this sort function to
142// general vectors of entries, it only works for matching results. Specifically
143// the entries are pruned to longest prefix matches.
144pub fn sort_entries<const D: usize, A: Clone>(
145    mut entries: Vec<TableEntry<D, A>>,
146) -> Vec<TableEntry<D, A>> {
147    if entries.is_empty() {
148        return entries;
149    }
150
151    // First determine how to sort. All the entries have the same keyset
152    // structure, so it's sufficient to iterate over the keys of the first
153    // entry. The basic logic we follow here is
154    //
155    // - If we find an lpm key sort on the prefix lenght of that dimension.
156    // - Otherwise simply sort on priority.
157    //
158    // Notes:
159    //
160    // It's assumed that multiple lpm keys do not exist in a single keyset.
161    // This is an implicit assumption in BVM2
162    //
163    //      https://github.com/p4lang/behavioral-model/issues/698
164    //
165    for (i, k) in entries[0].key.iter().enumerate() {
166        match k {
167            Key::Lpm(_) => {
168                let mut entries = prune_entries_by_lpm(i, &entries);
169                sort_entries_by_priority(&mut entries);
170                return entries;
171            }
172            _ => continue,
173        }
174    }
175
176    sort_entries_by_priority(&mut entries);
177    entries
178}

180// TODO - the data structures here are quite dumb. The point of this table
181// implemntation is not efficiency, it's a simple bruteish apprach that is easy
182// to move around until we nail down the, tbh, rather undefined (in terms of p4
183// spec)semantics of what the relative priorities between match types in a
184// common keyset are.
185pub fn prune_entries_by_lpm<const D: usize, A: Clone>(
186    d: usize,
187    entries: &Vec<TableEntry<D, A>>,
188) -> Vec<TableEntry<D, A>> {
189    let mut longest_prefix = 0u8;
190
191    for e in entries {
192        if let Key::Lpm(x) = &e.key[d] {
193            if x.len > longest_prefix {
194                longest_prefix = x.len
195            }
196        }
197    }
198
199    let mut result = Vec::new();
200    for e in entries {
201        if let Key::Lpm(x) = &e.key[d] {
202            if x.len == longest_prefix {
203                result.push(e.clone())
204            }
205        }
206    }
207
208    result
209}

211pub fn sort_entries_by_priority<const D: usize, A: Clone>(
212    entries: &mut [TableEntry<D, A>],
213) {
214    entries
215        .sort_by(|a, b| -> std::cmp::Ordering { b.priority.cmp(&a.priority) });
216}

218pub fn key_matches(selector: &BigUint, key: &Key) -> bool {
219    match key {
220        Key::Exact(x) => {
221            let hit = selector == &x.value;
222            if !hit {
223                let dump = format!("{:x} != {:x}", selector, x.value);
224                crate::p4rs_provider::match_miss!(|| &dump);
225            }
226            hit
227        }
228        Key::Range(begin, end) => {
229            let hit = selector >= &begin.value && selector <= &end.value;
230            if !hit {
231                let dump = format!(
232                    "begin={} end={} sel={}",
233                    begin.value, end.value, selector
234                );
235                crate::p4rs_provider::match_miss!(|| &dump);
236            }
237            hit
238        }
239        Key::Ternary(t) => match t {
240            Ternary::DontCare => true,
241            Ternary::Value(x) => selector == &x.value,
242            Ternary::Masked(x, m, _) => selector & m == x & m,
243        },
244        Key::Lpm(p) => match p.addr {
245            IpAddr::V6(addr) => {
246                assert!(p.len <= 128);
247                let key: u128 = addr.into();
248                let mask = if p.len == 128 {
249                    u128::MAX
250                } else if p.len == 0 {
251                    0u128
252                } else {
253                    ((1u128 << p.len) - 1) << (128 - p.len)
254                };
255                //let mask = mask.to_be();
256                let selector_v6 = selector.to_u128().unwrap();
257                let hit = selector_v6 & mask == key & mask;
258                if !hit {
259                    let dump = format!(
260                        "{:x} & {:x} == {:x} & {:x} | {:x} == {:x}",
261                        selector_v6,
262                        mask,
263                        key,
264                        mask,
265                        selector_v6 & mask,
266                        key & mask
267                    );
268                    //println!("{}", dump);
269                    crate::p4rs_provider::match_miss!(|| &dump);
270                }
271                hit
272            }
273            IpAddr::V4(addr) => {
274                assert!(p.len <= 32);
275                let key: u32 = addr.into();
276                let mask = if p.len == 32 {
277                    u32::MAX
278                } else if p.len == 0 {
279                    0
280                } else {
281                    ((1u32 << p.len) - 1) << (32 - p.len)
282                };
283                let selector_v4: u32 = selector.to_u32().unwrap();
284                let hit = selector_v4 & mask == key & mask;
285                if !hit {
286                    let dump = format!(
287                        "{:x} & {:x} == {:x} & {:x} | {:x} = {:x}",
288                        selector_v4,
289                        mask,
290                        key,
291                        mask,
292                        selector_v4 & mask,
293                        key & mask
294                    );
295                    crate::p4rs_provider::match_miss!(|| &dump);
296                }
297                hit
298            }
299        },
300    }
301}

303pub fn keyset_matches<const D: usize>(
304    selector: &[BigUint; D],
305    key: &[Key; D],
306) -> bool {
307    for i in 0..D {
308        if !key_matches(&selector[i], &key[i]) {
309            return false;
310        }
311    }
312    true
313}

315#[derive(Clone)]
316pub struct TableEntry<const D: usize, A: Clone> {
317    pub key: [Key; D],
318    pub action: A,
319    pub priority: u32,
320    pub name: String,
321
322    // the following are not used operationally, strictly for observability as
323    // the closure contained in `A` is hard to get at.
324    pub action_id: String,
325    pub parameter_data: Vec<u8>,
326}
327
328// TODO: Cannot hash on just the key, this does not work for multipath.
329impl<const D: usize, A: Clone> std::hash::Hash for TableEntry<D, A> {
330    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
331        self.key.hash(state);
332    }
333}

335impl<const D: usize, A: Clone> std::cmp::PartialEq for TableEntry<D, A> {
336    fn eq(&self, other: &Self) -> bool {
337        self.key == other.key
338    }
339}

341impl<const D: usize, A: Clone> std::fmt::Debug for TableEntry<D, A> {
342    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
343        f.debug_struct(&format!("TableEntry<{}>", D))
344            .field("key", &self.key)
345            .field("priority", &self.priority)
346            .field("name", &self.name)
347            .finish()
348    }
