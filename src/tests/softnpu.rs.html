11pub fn do_expect_frames(
12    name: &str,
13    phy: &Arc<OuterPhy<RING, FBUF, MTU>>,
14    expected: &[RxFrame],
15    dmac: Option<[u8; 6]>,
16) {
17    let n = expected.len();
18    let mut frames = Vec::new();
19    loop {
20        let fs = phy.recv();
21        frames.extend_from_slice(&fs);
22        // TODO this is not a great interface, if frames.len() > n, we should do
23        // something besides hang forever.
24        if frames.len() == n {
25            break;
26        }
27    }
28    for i in 0..n {
29        let payload = match frames[i].ethertype {
30            0x0901 => {
31                let mut payload = &frames[i].payload[..];
32                let et = u16::from_be_bytes([payload[5], payload[6]]);
33                payload = &payload[23..];
34                if et == 0x86dd {
35                    payload = &payload[40..];
36                }
37                if et == 0x0800 {
38                    payload = &payload[20..];
39                }
40                payload
41            }
42            0x86dd => &frames[i].payload[40..],
43            0x0800 => &frames[i].payload[20..],
44            _ => &frames[i].payload[..],
45        };
46        let m = String::from_utf8_lossy(payload).to_string();
47        println!("[{}] {}", name.magenta(), m.dimmed());
48        assert_eq!(frames[i].src, expected[i].src, "src");
49        if let Some(d) = dmac {
50            assert_eq!(frames[i].dst, d, "dst");
51        }
52        assert_eq!(frames[i].ethertype, expected[i].ethertype, "ethertype");
53        assert_eq!(payload, expected[i].payload, "payload");
54    }
55}

81pub struct SoftNpu<P: p4rs::Pipeline> {
82    pub pipeline: Option<P>,
83    inner_phys: Option<Vec<InnerPhy<RING, FBUF, MTU>>>,
84    outer_phys: Vec<Arc<OuterPhy<RING, FBUF, MTU>>>,
85    _fb: Arc<FrameBuffer<FBUF, MTU>>,
86}
87
88impl<P: p4rs::Pipeline + 'static> SoftNpu<P> {
89    /// Create a new SoftNpu ASIC emulator. The `radix` indicates the number of
90    /// ports. The `pipeline` is the `x4c` compiled program that the ASIC will
91    /// run. When `cpu_port` is set to true, sidecar data in `TxFrame` elements
92    /// will be added to packets sent through port 0 (as a sidecar header) on
93    /// the way to the ASIC.
94    pub fn new(radix: usize, pipeline: P, cpu_port: bool) -> Self {
95        let fb = Arc::new(FrameBuffer::<FBUF, MTU>::new());
96        let mut inner_phys = Vec::new();
97        let mut outer_phys = Vec::new();
98        for i in 0..radix {
99            let (rx_p, rx_c) = ring::<RING, FBUF, MTU>(fb.clone());
100            let (tx_p, tx_c) = ring::<RING, FBUF, MTU>(fb.clone());
101            let inner_phy = InnerPhy::new(i, rx_c, tx_p);
102            let mut outer_phy = OuterPhy::new(i, rx_p, tx_c);
103            inner_phys.push(inner_phy);
104            if i == 0 && cpu_port {
105                outer_phy.sidecar_encap = true;
106            }
107            outer_phys.push(Arc::new(outer_phy));
108        }
109        let inner_phys = Some(inner_phys);
110        SoftNpu {
111            inner_phys,
112            outer_phys,
113            _fb: fb,
114            pipeline: Some(pipeline),
115        }
116    }
117
118    pub fn run(&mut self) {
119        let inner_phys = match self.inner_phys.take() {
120            Some(phys) => phys,
121            None => panic!("phys already in use"),
122        };
123        let pipe = match self.pipeline.take() {
124            Some(pipe) => pipe,
125            None => panic!("pipe already in use"),
126        };
127        spawn(move || {
128            Self::do_run(inner_phys, pipe);
129        });
130    }
131
132    fn do_run(inner_phys: Vec<InnerPhy<RING, FBUF, MTU>>, mut pipeline: P) {
133        loop {
134            // TODO: yes this is a highly suboptimal linear gather-scatter across
135            // each ingress. Will update to something more concurrent eventually.
136            for (i, ig) in inner_phys.iter().enumerate() {
137                // keep track of how many frames we've produced for each egress
138                let mut egress_count = vec![0; inner_phys.len()];
139
140                // keep track of how many frames we've consumed for this ingress
141                let mut frames_in = 0;
142
143                for fp in ig.rx_c.consumable() {
144                    frames_in += 1;
145                    let content = ig.rx_c.read_mut(fp);
146
147                    let mut pkt = packet_in::new(content);
148
149                    let port = i as u16;
150                    let output = pipeline.process_packet(port, &mut pkt);
151                    for (out_pkt, out_port) in &output {
152                        let out_port = *out_port as usize;
153                        //
154                        // get frame for packet
155                        //
156
157                        let phy = &inner_phys[out_port];
158                        let eg = &phy.tx_p;
159                        let mut fps = eg.reserve(1).unwrap();
160                        let fp = fps.next().unwrap();
161
162                        //
163                        // emit headers
164                        //
165
166                        eg.write_at(fp, out_pkt.header_data.as_slice(), 0);
167
168                        //
169                        // emit payload
170                        //
171
172                        eg.write_at(
173                            fp,
174                            out_pkt.payload_data,
175                            out_pkt.header_data.len(),
176                        );
177
178                        egress_count[out_port] += 1;
179                    }
180                }
181                ig.rx_c.consume(frames_in).unwrap();
182                ig.rx_counter.fetch_add(frames_in, Ordering::Relaxed);
183
184                for (j, n) in egress_count.iter().enumerate() {
185                    if *n == 0 {
186                        continue;
187                    }
188                    let phy = &inner_phys[j];
189                    phy.tx_p.produce(*n).unwrap();
190                    phy.tx_counter.fetch_add(*n, Ordering::Relaxed);
191                }
192            }
193        }
194    }
195
196    pub fn phy(&self, i: usize) -> Arc<OuterPhy<RING, FBUF, MTU>> {
197        self.outer_phys[i].clone()
198    }
199}

400impl<const R: usize, const N: usize, const F: usize> OuterPhy<R, N, F> {
401    pub fn new(
402        index: usize,
403        rx_p: RingProducer<R, N, F>,
404        tx_c: RingConsumer<R, N, F>,
405    ) -> Self {
406        let mut rng = rand::rng();
407        let m = rng.random_range::<u32, _>(0xf00000..0xffffff).to_le_bytes();
408        let mac = [0xa8, 0x40, 0x25, m[0], m[1], m[2]];
409
410        Self {
411            index,
412            rx_p,
413            tx_c,
414            mac,
415            tx_counter: AtomicUsize::new(0),
416            rx_counter: AtomicUsize::new(0),
417            sidecar_encap: false,
418        }
419    }
420
421    pub fn send(&self, frames: &[TxFrame<'_>]) -> Result<(), xfr::Error> {
422        let n = frames.len();
423        let fps = self.rx_p.reserve(n)?;
424        for (i, fp) in fps.enumerate() {
425            let f = &frames[i];
426            self.rx_p.write_at(fp, f.dst.as_slice(), 0);
427            self.rx_p.write_at(fp, &self.mac, 6);
428            let mut off = 12;
429            if self.sidecar_encap {
430                self.rx_p
431                    .write_at(fp, 0x0901u16.to_be_bytes().as_slice(), off);
432                off += 2;
433                // sc_code = SC_FWD_FROM_USERSPACE
434                self.rx_p.write_at(fp, &[0u8], off);
435                off += 1;
436                // sc_ingress
437                let ingress = f.sc_egress;
438                self.rx_p
439                    .write_at(fp, ingress.to_be_bytes().as_slice(), off);
440                off += 2;
441                // sc_egress
442                let egress = f.sc_egress;
443                self.rx_p.write_at(fp, egress.to_be_bytes().as_slice(), off);
444                off += 2;
445                // sc_ether_type
446                self.rx_p.write_at(
447                    fp,
448                    f.ethertype.to_be_bytes().as_slice(),
449                    off,
450                );
451                off += 2;
452                // sc_payload
453                self.rx_p.write_at(fp, [0u8; 16].as_slice(), off);
454                off += 16;
455            } else if let Some(vid) = f.vid {
456                self.rx_p
457                    .write_at(fp, 0x8100u16.to_be_bytes().as_slice(), off);
458                off += 2;
459                self.rx_p.write_at(fp, vid.to_be_bytes().as_slice(), off);
460                off += 2;
461                self.rx_p.write_at(
462                    fp,
463                    f.ethertype.to_be_bytes().as_slice(),
464                    off,
465                );
466                off += 2;
467            } else {
468                self.rx_p.write_at(
469                    fp,
470                    f.ethertype.to_be_bytes().as_slice(),
471                    off,
472                );
473                off += 2;
474            }
475
476            self.rx_p.write_at(fp, f.payload, off);
477        }
478        self.rx_p.produce(n)?;
479        self.tx_counter.fetch_add(n, Ordering::Relaxed);
480        Ok(())
481    }
482
483    pub fn recv(&self) -> Vec<OwnedFrame> {
484        let mut buf = Vec::new();
485        loop {
486            for fp in self.tx_c.consumable() {
487                let b = self.tx_c.read(fp);
488                let mut et = u16::from_be_bytes([b[12], b[13]]);
489                let mut vid: Option<u16> = None;
490                let payload = if et == 0x8100 {
491                    let v = u16::from_be_bytes([b[14], b[15]]);
492                    et = u16::from_be_bytes([b[16], b[17]]);
493                    vid = Some(v);
494                    b[18..].to_vec()
495                } else {
496                    b[14..].to_vec()
497                };
498                let frame = OwnedFrame::new(
499                    b[0..6].try_into().unwrap(),
500                    b[6..12].try_into().unwrap(),
501                    et,
502                    vid,
503                    payload,
504                );
505                buf.push(frame);
506            }
507            if !buf.is_empty() {
508                break;
509            }
510        }
511        self.tx_c.consume(buf.len()).unwrap();
512        self.rx_counter.fetch_add(buf.len(), Ordering::Relaxed);
513
514        buf
515    }
516
517    pub fn recv_buffer_len(&self) -> usize {
518        self.tx_c.consumable().count()
519    }
520
521    pub fn tx_count(&self) -> usize {
522        self.tx_counter.load(Ordering::Relaxed)
523    }
524
525    pub fn rx_count(&self) -> usize {
526        self.rx_counter.load(Ordering::Relaxed)
